#include <iostream>

{

  {

    cout << "1 chetvert'\n-----\n";   
   }
   if ((ox<0) && (oy>0))

    cout << "2 chetvert'\n-----\n";   
   }

   {
    cout << "3 chetvert'\n-----\n";   
   }
   if ((ox>0) && (oy<0))
   {
    cout << "4 chetvert'\n-----\n";   
   }
  }
}
int main()
